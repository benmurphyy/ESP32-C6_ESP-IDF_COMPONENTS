45/*
46
47 Constant: font8x8_latin_tr
48 Contains an 90 degree transposed 8x8 font map for unicode points
49
50 U+0000 - U+00FF (latin)
51
52 To make it easy to use with SSD1306's GDDRAM mapping and API,
53 this constant is an 90 degree transposed.
54 The original version written by Marcel Sondaar is availble at:
55 https://github.com/dhepper/font8x8/blob/master/font8x8_latin.h
56
57 Conversion is done via following procedure:
58
59 for (int code = 0; code < 256; code++) {
60 uint8_t trans[8];
61 for (int w = 0; w < 8; w++) {
62 trans[w] = 0x00;
63 for (int b = 0; b < 8; b++) {
64 trans[w] |= ((font8x8_latin[code][b] & (1 << w)) >> w) << b;
65 }
66 }
67
68 for (int w = 0; w < 8; w++) {
69 if (w == 0) { printf(" { "); }
70 printf("0x%.2X", trans[w]);
71 if (w < 7) { printf(", "); }
72 if (w == 7) { printf(" }, // U+00%.2X (%c)\n", code, code); }
73 }
74 }
75*/
